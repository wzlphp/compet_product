```python
            {

                result["battlefield"].append(kw)

                result["long_tail"].append(kw)
            else:
                result["avoid"].append(kw)
```

```python
# scheduler/monitor.py
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from datetime import datetime
import asyncio

class CompetitorMonitor:
    """竞品监控调度器"""
    
    def __init__(self, db, collectors, notifier):
        self.db = db
        self.collectors = collectors
        self.notifier = notifier
        self.scheduler = AsyncIOScheduler()
    
    def start(self):
        """启动监控任务"""
        # 每日任务
        self.scheduler.add_job(
            self.daily_check,
            'cron',
            hour=6,
            minute=0
        )
        # 每周任务
        self.scheduler.add_job(
            self.weekly_check,
            'cron',
            day_of_week='mon',
            hour=8
        )
        self.scheduler.start()
    
    async def daily_check(self):
        """每日检查: 价格、BSR、库存、评价"""
        competitors = await self.db.get_monitored_asins()
        
        for asin in competitors:
            try:
                current = await self.collectors.fetch_product(asin)
                previous = await self.db.get_latest_snapshot(asin)
                
                alerts = self.compare_and_alert(current, previous)
                
                if alerts:
                    await self.notifier.send_alert(asin, alerts)
                
                await self.db.save_snapshot(asin, current)
                
            except Exception as e:
                logging.error(f"Monitor error for {asin}: {e}")
    
    def compare_and_alert(self, current: Dict, previous: Dict) -> List[Dict]:
        """对比并生成报警"""
```

## 三、数据库设计

### 3.1 核心表结构

```sql
-- 商品表
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) UNIQUE NOT NULL,
    title TEXT,
    brand VARCHAR(255),
    category_id INTEGER,
    parent_asin VARCHAR(10),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 价格历史表
CREATE TABLE price_history (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) REFERENCES products(asin),
    price DECIMAL(10,2),
    bsr INTEGER,
    review_count INTEGER,
    rating DECIMAL(2,1),
    recorded_at TIMESTAMP DEFAULT NOW()
);

-- 关键词表 (v3.0 新增)
CREATE TABLE keywords (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) REFERENCES products(asin),
    keyword VARCHAR(500),
    search_volume INTEGER,
    keyword_type VARCHAR(20),  -- organic/sponsored/brand
    organic_rank INTEGER,
    sponsored_rank INTEGER,
    click_share DECIMAL(5,4),
    recorded_at TIMESTAMP DEFAULT NOW()
);

-- 评论表 (v3.0 新增)
CREATE TABLE reviews (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) REFERENCES products(asin),
    review_id VARCHAR(50) UNIQUE,
    rating INTEGER,
    title TEXT,
    content TEXT,
    is_vp BOOLEAN,
    helpful_votes INTEGER,
    sentiment VARCHAR(20),  -- positive/negative/neutral
    sentiment_score DECIMAL(3,2),
    review_date DATE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 监控配置表 (v3.0 新增)
CREATE TABLE monitor_config (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) REFERENCES products(asin),
    monitor_type VARCHAR(50),
    frequency VARCHAR(20),
    alert_threshold DECIMAL(5,2),
    enabled BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 报警记录表 (v3.0 新增)
CREATE TABLE alerts (
    id SERIAL PRIMARY KEY,
    asin VARCHAR(10) REFERENCES products(asin),
    alert_type VARCHAR(50),
    old_value TEXT,
    new_value TEXT,
    change_percent DECIMAL(10,4),
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW()
);
```

## 八、合规与风险

1. **API 使用**: 优先使用官方/授权 API
2. **爬虫限制**:
